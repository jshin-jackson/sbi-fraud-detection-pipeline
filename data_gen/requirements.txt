# ※ 중요: pip download 는 반드시 클러스터와 동일한 OS/아키텍처인
#          RHEL 9.6 x86_64 Bastion 머신에서 실행해야 합니다.
#          macOS나 다른 OS에서 실행하면 플랫폼 불일치로 패키지를 찾지 못합니다.
#
# ※ PRD 정책: 클러스터 노드에서 yum install 금지.
#              빌드 도구(gcc 등)는 Bastion에서만 사용하고, wheel을 반입합니다.
#              PRD 노드에는 krb5 런타임 라이브러리가 CDP 이미지에 이미 포함되어 있습니다.
#
# [Step 1] Bastion 머신에서만 — gssapi wheel 빌드용 시스템 패키지 설치:
#   sudo yum install -y gcc python3-devel krb5-devel
#
# [Step 2] RHEL 9.6 Bastion 머신에서 (인터넷 연결 필요):
#   python3 -m venv /tmp/sbi-venv
#   source /tmp/sbi-venv/bin/activate
#   pip install --upgrade pip
#   pip download -r requirements.txt -d ./wheels/
#   tar cf sbi-wheels.tar wheels/
#   scp sbi-wheels.tar systest@<클러스터-호스트>:/tmp/
#
# [Step 3] 클러스터 노드(PRD)에서 오프라인 설치 — yum install 불필요:
#   python3 -m venv /tmp/sbi-venv
#   source /tmp/sbi-venv/bin/activate
#   cd /tmp && tar xf sbi-wheels.tar
#   pip install --no-index --find-links=./wheels/ -r requirements.txt
